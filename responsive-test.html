üîß Browser Console Verification

Run this code in the browser console on any CRM page to verify responsive classes:

// Check if responsive classes are applied correctly
const checkResponsiveClasses = () => {
  const results = {};
  
  // Check sidebar
  const sidebar = document.querySelector('[role="navigation"]')?.parentElement;
  if (sidebar) {
    results.sidebar = ['w-52', 'lg:w-56', 'xl:w-60'].every(cls => 
      sidebar.classList.contains(cls)
    );
  }
  
  // Check dashboard grid
  const grid = document.querySelector('.grid');
  if (grid) {
    results.dashboardGrid = ['grid-cols-1', 'lg:grid-cols-2', 'xl:grid-cols-4'].some(cls =>
      grid.classList.contains(cls)
    );
  }
  
  // Check table scrolling
  const tableContainer = document.querySelector('[role="table"]')?.parentElement;
  if (tableContainer) {
    results.tableScrolling = ['overflow-x-auto', 'scrollbar-thin'].every(cls =>
      tableContainer.classList.contains(cls)
    );
  }
  
  console.log('Responsive Classes Check:', results);
  return results;
};

checkResponsiveClasses();
